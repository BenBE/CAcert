	include_once("../includes/mysql.php");

// english (use the same for foreign language, too)

	$lines_EN = "";
	if (file_exists("thawte_EN.txt"))
	{
		$fp = fopen("thawte_EN.txt", "r");
		while(!feof($fp))
		{
			$line = trim(fgets($fp, 4096));
			$lines_EN .= wordwrap($line, 75, "\n")."\n";
		}
		fclose($fp);
	}

	$lines_DE = "";
	if (file_exists("thawte_DE.txt"))
	{
		$fp = fopen("thawte_DE.txt", "r");
		while(!feof($fp))
		{
			$line = trim(fgets($fp, 4096));
			$lines_DE .= wordwrap($line, 75, "\n")."\n";
		}
		fclose($fp);
	}
	
	$lines_NL = "";
	if (file_exists("thawte_NL.txt"))
	{
		$fp = fopen("thawte_NL.txt", "r");
		while(!feof($fp))
		{
			$line = trim(fgets($fp, 4096));
			$lines_NL .= wordwrap($line, 75, "\n")."\n";
		}
		fclose($fp);
	}
	
	$lines_FR = "";
	if (file_exists("thawte_FR.txt"))
	{
		$fp = fopen("thawte_FR.txt", "r");
		while(!feof($fp))
		{
			$line = trim(fgets($fp, 4096));
			$lines_FR .= wordwrap($line, 75, "\n")."\n";
		}
		fclose($fp);
	}
	
	$lines_ES = "";
	if (file_exists("thawte_ES.txt"))
	{
		$fp = fopen("thawte_ES.txt", "r");
		while(!feof($fp))
		{
			$line = trim(fgets($fp, 4096));
			$lines_ES .= wordwrap($line, 75, "\n")."\n";
		}
		fclose($fp);
	}
	
	$lines_RU = "";
	if (file_exists("thawte_RU.txt"))
	{
		$fp = fopen("thawte_RU.txt", "r");
		while(!feof($fp))
		{
			$line = trim(fgets($fp, 4096));
			$lines_RU .= wordwrap($line, 75, "\n")."\n";
		}
		fclose($fp);
	}

// read last used id
	$lastid = 0;
	if (file_exists("send_thawte_lastid.txt"))
	{
		$fp = fopen("send_thawte_lastid.txt", "r");
		$line = fgets($fp, 4096);
		$lastid = wordwrap($line, 75, "\n")."\n";
		fclose($fp);
	}
	
	$subject = "Change in points calculation";

	echo "ID now: $lastid\n";

	$query = "select `id`,`fname`,`lname`,`email`,`language` from `users` where `deleted` = 0 and `id` > '$lastid' order by `id`";

	$res = mysql_query($query);

	while($row = mysql_fetch_assoc($res))
	{
		$mailtxt = "Hello ".$row["fname"]." ".$row["lname"]."\n".$lines_EN."\n\n";
		switch ($row["language"])
		{
			case "de_DE":
			case "de":
				$mailtxt .= $lines_DE;
				break;
			
			case "nl_NL":
			case "nl":
				$mailtxt .= $lines_NL;
				break;
			
			case "fr_FR":
			case "fr":
				$mailtxt .= $lines_FR;
				break;
			
			case "es_ES":
			case "es":
				$mailtxt .= $lines_ES;
				break;
			
			case "ru_RU":
			case "ru":
				$mailtxt .= $lines_RU;
				break;
		}
		
		sendmail($row['email'], "[CAcert.org] $subject", $mailtxt, "mailing@cacert.org", "", "", "CAcert", "returns@cacert.org", "");
		
		$fp = fopen("send_thawte_lastid.txt", "w");
		fputs($fp, $row["id"]."\n");
		fclose($fp);
		
		echo "Sent mail to: ".$row["id"]."\n";
		
		sleep (1);
	}
?>
